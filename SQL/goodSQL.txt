CREATE TABLE [dbo].[IPs] (
    [IP_Id]			[int]			IDENTITY(1,1) PRIMARY KEY,
    [Ip_Address]	[varchar](50)	NOT NULL,
	[IP_wrongAtt]	[int]			DEFAULT 0,
    [IP_CreatedAt]	[datetime]		DEFAULT GETDATE(),
	[IP_Banned]		[bit]			DEFAULT 0,
);

CREATE TABLE [dbo].[Topic](
